1. Get the application URL by running these commands:
{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "key-aws-exporter.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Visit: http://$NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  Watch the status by running 'kubectl get svc -w {{ include "key-aws-exporter.fullname" . }}'
{{- else }}
  kubectl port-forward svc/{{ include "key-aws-exporter.fullname" . }} 8080:{{ .Values.service.port }}
  echo "Visit: http://127.0.0.1:8080"
{{- end }}
